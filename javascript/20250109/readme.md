### 연산자  //----------------------------------------------------------------------------------------------------

``` js

// 단항 산술 연산자
// +, -, *, /, %

let num = 1;
let num = 2;
let num3 = num + num2;

console.log(num3);

// 문자열과  숫자를 더하는경우
// "1" + 1 => 11  문자열 되어서 뒤에 붙입니다.
// 1 + 1 = 2 숫자
// 형변환
// 연산자를 공부할때 중요한것은 반환값(쉽게 이해를 돕기위해서 연산자를 사용한 위치에 생성되는 값)  과 데이터 타입 ( 연산자가 반환하는 데이터 타입)

// 비교 연산자
// 두 개 이상의 값을 비교해거 true 와 false 의 값을 반환하는 연산자

let num4 = 3;
let num5 = 4;

num4 == num5; // are they equal  같니?  3 와 4  같지 않아 false 를 반환 데이터 타입 bool

// 일반 값을 사용해고 괴지만 우리가 이후에 호출해거 가용하지 못하는 값이겠죠? 

4 == 4 // 같니? 4 와 4 는 같으니 true 를 반환 (return)

// 문자형 데이터와 숫자형 데이터를 비교하는 경우

"4" == 4;  // 데이트 타입은 비교하지 않고 값만 비교합니다 
// 같다 왜 같냐


"4" === 4;  /// === 는 데이터 타입까지 비교한다  , 4 4 는 false이로 나옵니다 나입가 다르기때문에
console.log ( num4 == num 5);

// 부정 연산자 (note equal to, denial, opposite)
// bool 타입의 값을 반전시키고 싶을때 반대 되는값이 필료한 경우
// 변수의 값을 화인할때
// 사용을 해야하는데 중일수 있으면 줄이면 가족성이 좀더 좋아진다
//

 // true 의 반대인 false 
console.log(!true);  false 반환
console.log(!false);  true 반환

// 비교 연산자와 같이 사용할때 
console.log(4 != 4); // 값이 다르니? 달랐을떄 true 으로 나옵니다

// 타입까지 비교 

console.log("3" == 3);  // 값하고 타입까지 같아?  타입이 다르기때문에 false 으로 나옵니다

// 큰 값과 작은 값을 비요하는 연산자

//  <,  >  : 크기를 비교하는 연산자. 초과와 미만을 비교할대  3 < 3    3 이 3 보다 크지 않으니 초과되지 않으니 false  3 < 4  4 는 3 보다 초과죈 값이기 때문에   true
//  <=,  => : 이상과 이하를 비교할때 이상과 이하  3 <= 3  같거나 큰값인지  true  3 <=  4 true

```

### 조건문

> 개발자의 꽃 조건문 
> 걷고 있는데 만보까지만 포인트를 줄거야 앱을 개발한다 하면
> 걷는 거리가 어느 정도 일대마다 포인트를 줄건지 ( 조건문)

``` js
// 조건문 문법
[if [예약어] (조건문) ]{
    조건이 충죽했을때 실행할 스코프 ( 영역)
}

// 실행이 안될때도 있는거냐?
// 조건문이 충족하지 않았을때
// true 인지 false 인지 조건이다 true


// bool true
if(3 < 4) {
    console.log('3 은 4 보다 커');
}

// 여러개를 추가해고 된다
// 구현이후에 클린 코드 (clean coding, fair recoding)

let str = "나는 이수호 학생이야"

if(str === "나는 이준현 학생이야") {
    //첫번쨰 조건이 충족했으면 실행하고 종류 
    // 조건이 충족하자 않았으면 다음번 조건 확인
}
else if(str === "나는 이수호 학생이야"){
    // 조건이 충족되면 실행하고 종류
}




```

### 반복문
> 반복문으로 실행되어야하는 코드를 작성할때
> 숫자를 1 2 3 4 5 6 7 8 증가시켜서 숫자를 출력하고 싶다

```js
let a = 0;
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
console.log(++a);
// 코드의 간결성, 가독성을 해치는 행위, 확장성


[for 예약어] (선언문; 조건식; 증감식;) {
    // 스코프 영역 코드 작성 영역
}

// 순서를 표현하기 위해서

for(let i = 0; i < 10; i++){    // if   i  is minute 
    for(let j =0; j< 10; j++){      // j is second
        console.log(i, j);
    }
    console.log(i);  // 원하는 인덱스만크 반복되어야할 코드
}
for(let i = 1; i <= 10; i++){
    console.log(i);
}
// let i = 0; 선언문 읽고
// i <10 조건문 확인
// 코드 영역을 실행하고
// i++ 증감식 실행
// 조건식 확인
// 코드 영역 실행
// 이걸 반복으로 조건문에 충족하지 않을때까지

// 선언문 -> 조건식 -> 코드실행 -> 증감식 -> 조건문 -> 코드실행 -> ( 증감식 -> 조건문 -> 코드실행 )조건문 충족하지 않을때까지반복

// 1증
// 2 증
// 3 증  쓰면 안됩니다 반복  X  X  X
```
